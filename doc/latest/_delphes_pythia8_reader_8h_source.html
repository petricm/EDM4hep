131 void fillParticle(int id, double pMax, double etaMax,
132  Pythia8::Event &event, Pythia8::ParticleData &pdt, Pythia8::Rndm &rndm)
133 {
134  // Reset event record to allow for new event.
135  event.reset();
136 
137  // Generate uniform pt and eta.
138  double pt, eta, phi, pp, ee, mm;
139 
140  // pMin = 0.1 GeV for single particles
141  pp = pow(10, -1.0 + (log10(pMax) + 1.0) * rndm.flat());
142  eta = (2.0 * rndm.flat() - 1.0) * etaMax;
143  phi = 2.0 * M_PI * rndm.flat();
144  mm = pdt.mSel(id);
145  ee = Pythia8::sqrtpos(pp * pp + mm * mm);
146  pt = pp / cosh(eta);
147 
148  // Store the particle in the event record.
149  event.append(id, 1, 0, 0, pt * cos(phi), pt * sin(phi), pt * sinh(eta), ee, mm);
150 }
